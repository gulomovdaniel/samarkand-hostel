        <script>
          // Простая клиентская обработка формы.
          function submitBooking(e){
            e.preventDefault();
            const name = document.getElementById('name').value.trim();
            const email = document.getElementById('email').value.trim();
            const phone = document.getElementById('phone').value.trim();
            const checkin = document.getElementById('checkin').value;
            const checkout = document.getElementById('checkout').value;
            const room = document.getElementById('room').value;
            const message = document.getElementById('message').value.trim();

            // Валидация дат
            if (new Date(checkout) <= new Date(checkin)){
              showMessage('Дата выезда должна быть позже даты заезда', true);
              return false;
            }

            // Здесь: отправка на email/сервер. Пока — показываем примерный текст для копирования/отправки.
            const body = 
`Новая заявка на бронирование:
Имя: ${name}
Email: ${email}
Телефон: ${phone}
Заезд: ${checkin}
Выезд: ${checkout}
Номер: ${room}
Примечание: ${message}`;

            // Для демонстрации копируем текст в буфер и показываем сообщение.
            navigator.clipboard?.writeText(body).then(() => {
              showMessage('Заявка подготовлена и скопирована в буфер обмена. Отправьте её нам в WhatsApp или по email (+998 66 123 45 67).', false);
            }).catch(() => {
              showMessage('Заявка готова. Скопируйте текст вручную:\n' + body, false);
            });

            // Если хочешь — можно отправить форму на Formspree или Getform. 
            return false;
          }

          function showMessage(text, isError){
            const el = document.getElementById('formMessage');
            el.style.color = isError ? 'crimson' : 'green';
            el.textContent = text;
          }

          function resetForm(){
            document.getElementById('bookingForm').reset();
            showMessage('', false);
          }
        </script>
